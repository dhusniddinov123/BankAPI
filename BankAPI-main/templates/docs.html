/send_code/
data class SendCode (phone_number: String)
POST ni Body qismiga Data Class yaratilib phone_number berilishi kerak
Natija:
data class Response (status: String)
status: "Verification code sent" yoki "Incorrect Phone Number"
POST

Kiritilgan kodni tekshirish

/check_code/
data class CheckCode (phone_number: String, code: String)
POST ni Body qismiga Data Class yaratilib phone_number va code berilishi kerak
Natija:
data class Response (status: String)
status: "Success", "Fail" yoki "Code not found"
POST

Kodni tekshirib karta qoshish

/add_card/
data class CheckCode (phone_number: String, code: String, card_number: String)
POST ni Body qismiga Data Class yaratilib phone_number, code va card_number berilishi kerak
Natija:
data class Response (status: String, card: Card)
status: "Success", "Fail" yoki "Code or Card not found"
POST

Tranzaksiya amalga oshirish

/transaction/
data class Transaction (sender: String, receiver: String, amount: Int)
POST ni Body qismiga Data Class yaratilib sender, receiver va amount berilishi kerak
Natija:
data class Response (status: String)
status: "Success", "Sender Does Not Have Enough Funds" yoki "Sender or Receiver Card not found or Invalid amount"
